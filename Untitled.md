SELECT *, AVG(Hours) 
FROM EMPLOYEE JOIN WORKS_ON 
WHERE EMPLOYEE.SSN = WORKS_ON.ESSN 
AND Pno IN (
	SELECT Pno FROM PROJECT 
	WHERE Pnumber=Pno 
	AND Dnum IN 
		(SELECT Dnumber FROM DEPARTMENT
		WHERE Dnumber=Dnum
		AND Dnumber IN
			(SELECT Dnumber FROM DEPT_LOCATIONS 
			WHERE Dnumber=Dnumber 
			AND Dlocation = Plocation)
		)
);

SELECT * FROM EMPLOYEE 
WHERE NOT EXIST (
	SELECT * FROM PROJECT 
	WHERE Dnum=Dno
	WHERE Pnumber IN (
		SELECT Pno FROM WORKS_ON 
		WHERE Pno = Pnumber
		AND ESSN!=SSN
	)
)

	WHERE SUPERVISEE.SSN IN 
		(SELECT ESSN FROM WORKS_ON
		WHERE Pno IN (
			SELECT Pno FROM WORKS_ON AS WORKS_ON_2
			WHERE WORKS_ON_2.ESSN = EMPLOYEE.SSN
			AND WORKS_ON.ESSN != SUPERVISEE.SSN
		))
	)
